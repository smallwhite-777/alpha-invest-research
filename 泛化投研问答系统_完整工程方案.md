# 泛化投研问答系统 - 完整工程方案

> 适用于任何行业、任何上市公司的智能投研问答系统

---

## 一、系统整体架构

```
用户查询 → 意图识别 → 两阶段搜索 → Prompt构建 → LLM推理 → 结果格式化 → 用户
    ↓           ↓            ↓            ↓           ↓            ↓
意图类型   文件级匹配    System Prompt  云端API     结构化输出     标注来源
实体提取   内容级抽取    User Prompt   上下文      Markdown       引用列表
关键词    限流保护      格式约束      联网补充    表格/图表      元数据
```

**核心技术原则：**
1. 不依赖向量数据库，内存占用 < 2GB
2. 两阶段搜索：文件名匹配（快）→ 内容抽取（准）
3. 本地知识库为主，联网补充为辅
4. 32GB内存笔记本可运行，无GPU需求

## 二、意图识别模块（IntentAgent）

### 2.1 核心职责
将用户模糊的自然语言转换为结构化的查询指令，包含：
- **意图类型**：公司分析、行业分析、对比分析、趋势预测、财务分析、估值分析、风险评估、一般问答
- **实体识别**：公司名称、股票代码、行业关键词
- **搜索关键词**：用于知识库检索的关键词列表
- **时间范围**：用户提到的具体年份或时间范围
- **置信度**：识别结果的可靠性评分

## 三、两阶段搜索模块（KnowledgeBaseSearcher）

### 3.1 设计原理

**核心思想：不读文件，只读文件名**

传统方案：读取文件 → 建立向量 → 语义检索
我们的方案：文件名匹配 → 选择性读取 → 内容抽取

**优势：**
- 索引速度：O(n) 目录扫描 vs O(n) 向量计算
- 内存占用：< 2MB 映射表 vs > 16GB 向量空间
- 精准度：100% 公司匹配 vs 语义漂移

### 3.3 Phase 1：文件名级过滤

```python
def _phase1_filter_files(self, keywords: List[str]) -> List[Dict]:
    """
    策略：只扫描目录名/文件名，绝对不读取文件内容
    优先级：完全匹配 > 部分匹配 > 模糊匹配
    """
    candidates = []
    
    # 1. 从公司目录名匹配
    for company_dir in os.listdir(RESEARCH_REPORTS_DIR):
        dir_lower = company_dir.lower()
        score = 0
        matched_keywords = []
        
        for kw in keywords:
            if kw in dir_lower:
                score += 10  # 基础分
                matched_keywords.append(kw)
                
                # 精确匹配公司全称（最高优先级）
                if dir_lower.startswith(kw.lower()):
                    score += 20
                
                # 匹配股票代码
                if kw.isdigit() and kw in company_dir:
                    score += 15
        
        if matched_keywords:
            candidates.append({
                'path': company_dir,
                'score': score,
                'matched': matched_keywords
            })
    
    # 2. 从文件名匹配
    for news_file in os.listdir(NEWS_DIR):
        for kw in keywords:
            if kw.lower() in news_file.lower():
                candidates.append({
                    'path': news_file,
                    'score': 5,  # 新闻文件权重较低
                    'type': 'news'
                })
    
    # 3. 排序返回
    return sorted(candidates, key=lambda x: x['score'], reverse=True)
```

### 3.4 Phase 2：内容抽取（有限读取）

```python
# 安全限制配置
MAX_CANDIDATE_FILES = 15        # 最多读取15个文件
MAX_READ_CHARS = 8000           # 单文件最多读取8KB
SEARCH_TIMEOUT = 30             # 搜索超时30秒
FILE_READ_TIMEOUT = 5           # 单文件读取超时5秒

def _read_file_with_timeout(self, file_path: str) -> str:
    """带超时保护的有限读取"""
    try:
        with open(file_path, 'r', encoding='utf-8', errors='ignore') as f:
            # 限制读取大小
            content = f.read(MAX_READ_CHARS)
        
        # 提取关键段落
        return self._extract_key_content(content)
    except Exception as e:
        return f"[读取失败: {str(e)}]"

def _extract_key_content(self, content: str) -> str:
    """提取核心内容段落"""
    # 1. 优先提取摘要/核心观点
    patterns = [
        r"(?:核心观点|投资要点|摘要|Summary|核心逻辑)[:：\s]*\n(.*?)(?=\n#{1,3}|\Z)",
        r"(?:风险提示|主要风险)[:：\s]*\n(.*?)(?=\n#{1,3}|\Z)",
        r"(?:财务数据|关键指标)[:：\s]*\n(.*?)(?=\n#{1,3}|\Z)",
    ]
    
    for pattern in patterns:
        match = re.search(pattern, content, re.DOTALL | re.IGNORECASE)
        if match:
            return match.group(1).strip()
    
    # 2. 返回前N个字符
    return content[:MAX_READ_CHARS]
```

```python
def _build_user_prompt(self, user_query: str, context: str, intent_result) -> str:
    """构建用户Prompt"""
    
    # 上下文格式化
    context_label = {
        IntentType.COMPANY_ANALYSIS: "公司深度研究报告",
        IntentType.INDUSTRY_ANALYSIS: "行业研究报告",
        IntentType.COMPARISON: "对比公司研究报告",
        IntentType.TREND_PREDICTION: "趋势分析资料",
        IntentType.FINANCIAL_ANALYSIS: "财务分析数据",
        IntentType.VALUATION: "估值分析资料",
        IntentType.RISK_ASSESSMENT: "风险评估资料",
        IntentType.GENERAL_QUESTION: "相关研究资料"
    }.get(intent_result.intent, "相关资料")
    
    # 时间范围注入
    time_context = ""
    if intent_result.time_range:
        time_context = f"\n\n【时间范围】用户关注的时间段是 {intent_result.time_range}，请优先使用该时段数据。"
    
    # 无数据时的处理
    if not context or context.strip() == "":
        context_section = """
【注意】本地知识库中未找到直接相关的资料。
请基于你的专业知识进行回答，并在回答中明确说明：
1. 哪些是基于已有知识的分析
2. 哪些信息需要用户自行验证
3. 建议用户补充哪些数据"""
    else:
        context_section = f"""
【本地知识库资料】
---
{context}
---

请特别关注上述资料中的数据，在回答中引用并验证。{time_context}
"""
    
    # 最终Prompt
    user_prompt = f"""{context_section}

【用户问题】
{user_query}

【回答要求】
1. 首先回答用户的核心问题
2. 提供详细的分析和数据支撑
3. 标注信息来源（来自本地知识库 or 基于公开知识）
4. 给出明确的结论和建议
5. 如信息不足，说明局限性

请开始回答："""

    return user_prompt
```

```python
def chat_with_retry(self, prompt, max_retries=3):
    """带指数退避的重试"""
    for attempt in range(max_retries):
        try:
            return self._call_llm(prompt)
        except RateLimitError:
            wait = 2 ** attempt * 10  # 10s, 20s, 40s
            time.sleep(wait)
        except ServerError:
            wait = 2 ** attempt * 5   # 5s, 10s, 20s
            time.sleep(wait)
        except NetworkError:
            wait = 5
            time.sleep(wait)
    raise Exception("Max retries exceeded")
```

```python
def _add_citations(self, response: str, sources: List[Dict]) -> str:
    """在回答末尾添加来源标注"""
    
    if not sources:
        footer = """
---
*注：本地知识库中未找到相关参考资料，当前回答基于公开知识整理*
"""
    else:
        citation_lines = ["\n\n---\n**信息来源：**\n"]
        
        for i, source in enumerate(sources, 1):
            parts = []
            if source.get("company_name"):
                parts.append(source["company_name"])
            if source.get("stock_code"):
                parts.append(f"({source['stock_code']})")
            if source.get("broker"):
                parts.append(f"- {source['broker']}")
            if source.get("date"):
                parts.append(f"({source['date']})")
            
            citation_lines.append(f"{i}. {' '.join(parts)}")
        
        footer = "".join(citation_lines)
    
    return response + footer
```
